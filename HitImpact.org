* About

HitImpact spawns textured particles when a wall, a floor, or a ceiling is hit.

Features:
- works for hitscan and projectile attacks,
- works for all games and mods (in theory),
- works even when loaded with old saves.

Known issues:
- Performance. HitImpact tries to minimize the amount of particles created when a lot
  of hits happen in a short period of time, but performance drop is still inevitable.
- Silliness. Basically, an impact particle has a miniature version of
  wall/floor/ceiling texture, and for some textures this may look silly.

Todo:
- 3D floor support.

class hi_EventHandler : StaticEventHandler
{
  override void worldThingDestroyed(WorldEvent event)
  {
    int found = mMissiles.find(event.thing);
    if (found != mMissiles.size()) mMissiles[found] = NULL;
  }

  override void worldTick()
  {
    int missileCount = mMissiles.size();
    for (int i = 0; i < missileCount; ++i)
    {
      Actor missile = mMissiles[i];
      if (!missile || missile.vel != mNullVector) continue;

      spawnDebrisMissile(missile.target, missile, 3);
      mMissiles[i] = NULL;
    }

    // Clean up.
    if ((level.time % TICRATE) == 0 && missileCount > 0)
    {
      mNewMissiles.clear();
      foreach (missile : mMissiles)
        if (missile) mNewMissiles.push(missile);

      mMissiles.move(mNewMissiles);
    }

    mAmountFactor = max(0, mAmountFactor - 0.5);
  }

  private Array<Actor> mMissiles;
  private Array<Actor> mNewMissiles;
  private double mAmountFactor;

  override void worldThingSpawned(WorldEvent event)
  {
    Actor thing = event.thing;
    if (thing)
    {
      if (thing.bMissile && thing.damage)
        mMissiles.push(thing);

      else if (thing.bIsPuff && thing.damageSource)
        spawnDebrisPuff(thing.damageSource, thing, 1);
    }
  }

  override void onRegister()
  {
    mParticleParameters.color1     = "";
    mParticleParameters.style      = STYLE_Normal;
    mParticleParameters.flags      = SPF_FaceCamera | SPF_Roll;
    mParticleParameters.startAlpha = START_ALPHA;
    mParticleParameters.accel      = (0, 0, -0.15);

    mNullVector = (0, 0, 0);

    for (int i = 0; i < CANVAS_COUNT; ++i)
    {
      string canvasName  = string.format("hi_texture%d", i);
      mCanvases[i]       = TexMan.getCanvas(canvasName);
      mCanvasTextures[i] = TexMan.checkForTexture(canvasName);
    }
  }

  private FSpawnParticleParams mParticleParameters;
  private vector3 mNullVector;
  const FLAGS = TRF_NoSky;
  const START_ALPHA = 1.0;

  private void spawnDebrisPuff(Actor source, Actor puff, double intensity)
  {
    FLineTraceData traceData;
    double angle = source.angleTo(puff);
    double offsetZ = getShootOffsetZ(source);
    double pitch = source.pitchTo(puff, offsetZ);
    bool   isHit = source.lineTrace(angle, 999, pitch, FLAGS, offsetZ, data: traceData);

    if (!isHit || traceData.hitType == FLineTraceData.TRACE_HitNone) return;
    if (!traceData.hitTexture) return;

    vector3 normal = getNormalFromTrace(traceData);
    if (normal == mNullVector) return;

    Color color = getColorFromTrace(traceData);
    spawnDebris(normal,
                traceData.hitDir,
                traceData.hitLocation,
                traceData.hitTexture,
                intensity,
                color);
  }

  private void spawnDebrisMissile(Actor source, Actor missile, double intensity)
  {
    vector3 normal = getNormalFromActor(missile);
    if (normal == mNullVector) return;

    vector3 hitDir = level.vec3Diff(source.pos, missile.pos).unit();
    TextureId texture = getTextureFromActor(missile);
    if (!texture) return;

    Color color = getColorFromActor(missile);
    spawnDebris(normal, hitDir, missile.pos, texture, intensity, color);
  }

  private void spawnDebris(vector3 normal,
                           vector3 incidence,
                           vector3 hitLocation,
                           TextureId texture,
                           double intensity,
                           Color color)
  {
    vector3 reflected = incidence - 2 * normal * (normal dot incidence);
    int     amount    = max(1, int(10 * intensity / (mAmountFactor + 1)));

    mParticleParameters.lifetime = int(TICRATE * intensity);
    mParticleParameters.fadeStep = START_ALPHA / mParticleParameters.lifetime;
    mParticleParameters.pos      = hitLocation;
    mParticleParameters.color1   = color;

    let [textureWidth, textureHeight] = TexMan.getSize(texture);
    int minDrawTextureX = CANVAS_SIZE - textureWidth;
    int minDrawTextureY = CANVAS_SIZE - textureHeight;

    for (int i = 0; i < amount; ++i)
    {
      mCanvasIndex = (mCanvasIndex + 1) % CANVAS_COUNT;
      mCanvases[mCanvasIndex].drawTexture(texture,
                                          false,
                                          random[hi](minDrawTextureX, 0),
                                          random[hi](minDrawTextureY, 0));

      mParticleParameters.texture = mCanvasTextures[mCanvasIndex];
      mParticleParameters.size = frandom[hi](1, 4 * intensity);
      mParticleParameters.vel  = reflected * 2 * intensity
        + (frandom[hi](-intensity, intensity),
           frandom[hi](-intensity, intensity),
           frandom[hi](-intensity, intensity));
      mParticleParameters.startRoll = frandom[hi](0, 360);
      mParticleParameters.rollVel   = frandom[hi](30, 60);

      level.spawnParticle(mParticleParameters);
    }

    if (amount > 1) mAmountFactor += 3;
  }

  private int mCanvasIndex;
  private Canvas mCanvases[CANVAS_COUNT];
  private TextureId mCanvasTextures[CANVAS_COUNT];
  const CANVAS_COUNT = <<canvas-count()>>;
  const CANVAS_SIZE = 12;

  // TODO: add 3D floors support.
  private static vector3 getNormalFromTrace(FLineTraceData data)
  {
    switch (data.hitType)
    {
      case FLineTraceData.TRACE_HitFloor:   return data.hitSector.floorPlane.normal;
      case FLineTraceData.TRACE_HitCeiling: return data.hitSector.ceilingPlane.normal;
      case FLineTraceData.TRACE_HitWall:
        return getNormalFromLine(data.hitLine, data.lineSide);
    }

    return (0, 0, 0);
  }

  // TODO: add 3D floors support.
  private static color getColorFromTrace(FLineTraceData data)
  {
    switch (data.hitType)
    {
      case FLineTraceData.TRACE_HitFloor:
        return data.hitSector.specialColors[Sector.floor];
      case FLineTraceData.TRACE_HitCeiling:
        return data.hitSector.specialColors[Sector.ceiling];
    }

    return Color("");
  }

  // TODO: add 3D floors support.
  private static vector3 getNormalFromActor(Actor actor)
  {
    if (actor.blockingFloor)   return actor.blockingFloor.floorPlane.normal;
    if (actor.blockingCeiling) return actor.blockingCeiling.ceilingPlane.normal;
    if (actor.blockingLine)
      return getNormalFromLine(actor.blockingLine, Line.front);

    return (0, 0, 0);
  }

  // TODO: add 3D floors support.
  private static TextureId getTextureFromActor(Actor actor)
  {
    if (actor.blockingFloor)   return actor.blockingFloor.getTexture(Sector.floor);
    if (actor.blockingCeiling) return actor.blockingCeiling.getTexture(Sector.ceiling);
    if (actor.blockingLine)
    {
      FLineTraceData traceData;
      vector2 lineCenter = (actor.blockingLine.v1.p + actor.blockingLine.v2.p) / 2;
      double offsetZ = actor.height / 2;
      vector2 dir = lineCenter - actor.pos.xy;
      double angle = atan2(dir.y, dir.x);
      bool   isHit = actor.lineTrace(angle, 999, 0, FLAGS, offsetZ, data: traceData);

      if (!isHit || traceData.hitType == FLineTraceData.TRACE_HitNone) return NULL;
      return traceData.hitTexture;
    }

    return NULL;
  }

  private static Color getColorFromActor(Actor actor)
  {
    if (actor.blockingFloor)
      return actor.blockingFloor.specialColors[Sector.floor];
    if (actor.blockingCeiling)
      return actor.blockingCeiling.specialColors[Sector.ceiling];

    return Color("");
  }

  private static vector3 getNormalFromLine(Line line, int side)
  {
    vector2 diff = (side == Line.front)
      ? line.v2.p - line.v1.p
      : line.v1.p - line.v2.p;

    return (diff.y, -diff.x, 0).unit();
  }

  private static double getShootOffsetZ(Actor actor)
  {
    let pawn = (PlayerPawn)(actor);
    return actor.height * 0.5
      + (pawn ? pawn.attackZOffset * pawn.player.crouchFactor - pawn.floorclip : 0);
  }
}
