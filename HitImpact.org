class hi_EventHandler : StaticEventHandler
{
  override void worldThingDestroyed(WorldEvent event)
  {
    int found = mMissiles.find(event.thing);
    if (found != mMissiles.size()) mMissiles.delete(found);
  }

  override void worldTick()
  {
    foreach (missile : mMissiles)
    {
      if (missile.vel != (0, 0, 0)) return;

      spawnDebris(missile.target, missile, 3);

      int found = mMissiles.find(missile);
      if (found != mMissiles.size()) mMissiles.delete(found);
    }
  }

  private Array<Actor> mMissiles;

  override void worldThingSpawned(WorldEvent event)
  {
    if (event.thing && event.thing.bMissile)
    {
      mMissiles.push(event.thing);
      return;
    }

    Actor puff = event.thing;
    if (!puff || !puff.bIsPuff || !puff.damageSource) return;

    spawnDebris(puff.damageSource, puff, 1);
  }

  private static void spawnDebris(Actor source, Actor impact, int intensity)
  {
    FLineTraceData traceData;
    double angle = source.angleTo(impact);
    double offsetZ = getShootOffsetZ(source);
    double pitch = source.pitchTo(impact, offsetZ);
    int    flags = TRF_NoSky;
    bool   isHit = source.lineTrace(angle, 999, pitch, flags, offsetZ, data: traceData);

    int hitType = traceData.hitType;
    if (!isHit || hitType == FLineTraceData.TRACE_HitNone) return;
    if (!traceData.hitTexture) return;

    // TODO: add 3D floors support. All directions?
    vector3 normal = getNormal(traceData);

    if (normal == (0, 0, 0)) return;
    vector3 incidence = traceData.hitDir;
    vector3 reflected = incidence - 2 * normal * (normal dot incidence);
    double velSpread = intensity;

    for (int i = 0; i < 10 * intensity; ++i)
    {
      FSpawnParticleParams parameters;
      parameters.color1  = "";
      parameters.texture = traceData.hitTexture;
      parameters.style   = STYLE_Normal;
      parameters.size    = frandom[hi](1, 4 * intensity);
      parameters.flags   = SPF_FaceCamera | SPF_Roll;

      parameters.lifetime   = TICRATE * intensity;
      parameters.startAlpha = 1.0;
      parameters.fadeStep   = parameters.startAlpha / parameters.lifetime;

      parameters.pos = traceData.hitLocation;
      parameters.vel = reflected * 2 * intensity
        + (frandom[hi](-velSpread, velSpread),
           frandom[hi](-velSpread, velSpread),
           frandom[hi](-velSpread, velSpread));
      parameters.accel = (0, 0, -0.15);

      parameters.startRoll = frandom[hi](0, 360);
      parameters.rollVel   = frandom[hi](30, 60);

      level.spawnParticle(parameters);
    }
  }

  private static vector3 getNormal(FLineTraceData data)
  {
    switch (data.hitType)
    {
      case FLineTraceData.TRACE_HitFloor:   return data.hitSector.floorPlane.normal;
      case FLineTraceData.TRACE_HitCeiling: return data.hitSector.ceilingPlane.normal;
      case FLineTraceData.TRACE_HitWall:
      {
        Line    line = data.hitLine;
        vector2 diff = (data.lineSide == Line.front)
          ? line.v2.p - line.v1.p
          : line.v1.p - line.v2.p;

        return (diff.y, -diff.x, 0).unit();
      }
    }

    return (0, 0, 0);
  }

  private static double getShootOffsetZ(Actor actor)
  {
    let pawn = (PlayerPawn)(actor);
    return actor.height * 0.5
      + (pawn ? pawn.attackZOffset * pawn.player.crouchFactor - pawn.floorclip : 0);
  }
}
