** ~TargetSpyClass~

TargetSpy can be extended by declaring classes as ~TargetSpyClass~. These classes are
not meant to be instantiated, only their static functions are used.

To be detected by TargetSpy, a class must:
- inherit Actor,
- have ~TargetSpyClass~ in its name,
- have a specific function, see below.

There are four types of ~TargetSpyClass~:
1. Finders find targets. They must have ~static void find(out ts_Target)~ function.
2. Filters remove unwanted targets. They must have ~static bool isAllowed(ts_Target)~
   function.
3. Informers collect information about a target. They must have ~static void
   makeInformation(ts_Target, out ts_InformationPiece)~ function.
4. Views display information. They must have have ~static ui void draw(out
   Array<ts_InformationPiece>)~ function.

~TargetSpyClass~ classes can have a translatable user string defined in ~LANGUAGE~
lump. The string key is simply a class name.

** Event handler

#+begin_src c :tangle build/TargetSpy/zscript/ts_event_handler.zs

class ts_EventHandler : StaticEventHandler
{
  override void onRegister()
  {
    collectClasses();
  }

  override void worldTick()
  {
    // TODO: reset mCurrentTarget on map change.
    // TODO: add titlemap check.
    if (players[consolePlayer].mo == NULL) return;

    findTargets();
    filterTargets();
    chooseCurrentTarget();
    collectInformation();
  }

  override void renderOverlay(RenderEvent event)
  {
    if (players[consolePlayer].mo == NULL) return;

    int viewsCount = mViews.size();
    for (int i = 0; i < viewsCount; ++i)
    {
      if (ts_ClassToggle.isEnabled(mViewClasses.classes[i]))
        mViews[i].call(mInformationPieces);
    }
  }

  static clearscope ts_EventHandler getInstance()
  {
    return (ts_EventHandler)(StaticEventHandler.find("ts_EventHandler"));
  }

  clearscope ts_Classes getFinderClasses()   { return mFinderClasses; }
  clearscope ts_Classes getFilterClasses()   { return mFilterClasses; }
  clearscope ts_Classes getInformerClasses() { return mInformerClasses; }
  clearscope ts_Classes getViewClasses()     { return mViewClasses; }

  private void findTargets()
  {
    int findersCount = mFinders.size();
    for (int i = 0; i < findersCount; ++i)
    {
      mTargets[i].anActor = NULL;
      if (ts_ClassToggle.isEnabled(mFinderClasses.classes[i]))
        mFinders[i].call(mTargets[i]);
    }
  }

  private void filterTargets()
  {
    int filtersCount = mFilters.size();

    foreach (target : mTargets)
    {
      if (target.isEmpty()) continue;

      for (int i = 0; i < filtersCount; ++i)
      {
        if (!ts_ClassToggle.isEnabled(mFilterClasses.classes[i])) continue;

        if (!mFilters[i].call(target))
        {
          target.clear();
          break;
        }
      }
    }
  }

  private void chooseCurrentTarget()
  {
    ts_Target closestTarget = NULL;
    double minDistance = double.max;
    foreach (target : mTargets)
    {
      if (target.isEmpty()) continue;
      double distance = distanceTo(target);
      if (distance < minDistance)
      {
        closestTarget = target;
        minDistance = distance;
      }
    }

    // TODO: keep old mCurrentTarget until timeout expires.
    mCurrentTarget = closestTarget;
  }

  private void collectInformation()
  {
    int informersCount = mInformers.size();
    for (int i = 0; i < informersCount; ++i)
      mInformationPieces[i].kind = ts_InformationPiece.KindEmpty;

    if (mCurrentTarget == NULL) return;

    for (int i = 0; i < informersCount; ++i)
    {
      if (ts_ClassToggle.isEnabled(mInformerClasses.classes[i]))
        mInformers[i].call(mCurrentTarget, mInformationPieces[i]);
    }

    // TODO: add priorities, sort information pieces according to priorities.
  }

  private void collectClasses()
  {
    mFinderClasses   = new("ts_Classes");
    mFilterClasses   = new("ts_Classes");
    mInformerClasses = new("ts_Classes");
    mViewClasses     = new("ts_Classes");

    foreach (aClass : AllActorClasses)
    {
      string className = aClass.getClassName();
      if (className.indexOf("TargetSpyClass") == -1) continue;

      if (!tryAddFinder(aClass)
          && !tryAddFilter(aClass)
          && !tryAddInformer(aClass)
          && !tryAddView(aClass))
        Console.printf("%s: unknown TargetSpyClass", className);
    }
  }

  private bool tryAddFinder(class<Object> aClass)
  {
    let finder = (Function<play void(out ts_Target)>)(findFunction(aClass, "find"));
    if (finder != NULL)
    {
      mFinders.push(finder);
      mFinderClasses.classes.push(aClass.getClassName());
      mTargets.push(new("ts_Target"));
    }
    return finder != NULL;
  }

  private bool tryAddFilter(class<Object> aClass)
  {
    let filter = (Function<play bool(ts_Target)>)(findFunction(aClass, "isAllowed"));
    if (filter != NULL)
    {
      mFilters.push(filter);
      mFilterClasses.classes.push(aClass.getClassName());
    }
    return filter != NULL;
  }

  private bool tryAddInformer(class<Object> aClass)
  {
    let informer = (Function<play void(ts_Target, out ts_InformationPiece)>)
      (findFunction(aClass, "makeInformation"));
    if (informer != NULL)
    {
      mInformers.push(informer);
      mInformerClasses.classes.push(aClass.getClassName());
      mInformationPieces.push(new("ts_InformationPiece"));
    }
    return informer != NULL;
  }

  private bool tryAddView(class<Object> aClass)
  {
    let view = (Function<ui void(Array<ts_InformationPiece>)>)
      (findFunction(aClass, "draw"));
    if (view != NULL)
    {
      mViews.push(view);
      mViewClasses.classes.push(aClass.getClassName());
    }
    return view != NULL;
  }

  private static double distanceTo(ts_Target target)
  {
    if (target.anActor != NULL)
      return players[consolePlayer].mo.distance3d(target.anActor);

    return 0;
  }

  Array< Function<play void(out ts_Target)> > mFinders;
  ts_Classes mFinderClasses;
  Array<ts_Target> mTargets;

  Array< Function<play bool(ts_Target)> > mFilters;
  ts_CLasses mFilterClasses;

  ts_Target mCurrentTarget;

  Array< Function<play void(ts_Target, out ts_InformationPiece)> > mInformers;
  ts_Classes mInformerClasses;
  Array<ts_InformationPiece> mInformationPieces;

  Array< Function<ui void(Array<ts_InformationPiece>)> > mViews;
  ts_Classes mViewClasses;
}

// TODO: rename this class.
class ts_ActorFinderTargetSpyClass : Actor
{
  static void find(out ts_Target target)
  {
    PlayerInfo player = players[consolePlayer];
    PlayerPawn pawn = player.mo;

    // Try an easy way to get a target (also works with autoaim).
    Actor aimTarget = pawn.aimTarget();

    if (aimTarget != NULL)
    {
      target.anActor = aimTarget;
      return;
    }

    // If target is not found by easy way, try the difficult way.
    FLineTraceData lineTraceData;
    double angle = pawn.angle;
    double pitch = pawn.pitch;
    double viewHeight = player.viewHeight;
    // TODO: move 4000.0 to a constant?
    pawn.lineTrace(angle, 4000.0, pitch, 0, viewHeight, 0.0, 0.0, lineTraceData);

    target.anActor = lineTraceData.hitActor;
  }
}

class tt_TextViewTargetSpyClass : Actor
{
  static ui void draw(Array<ts_InformationPiece> informationPieces)
  {
    Font aFont = Font.findFont("NewSmallFont");
    int xCenter = Screen.getWidth() / 2;
    int y = int(round(Screen.getHeight() * 0.2));
    int lineHeight = aFont.getHeight();

    foreach (informationPiece : informationPieces)
    {
      if (informationPiece.kind == ts_InformationPiece.KindValue)
      {
        drawText(string.format("%d", informationPiece.value), xCenter, y);
        y += lineHeight;
      }
      else if (informationPiece.kind == ts_InformationPiece.KindPlainText)
      {
        drawText(informationPiece.text, xCenter, y);
        y += lineHeight;
      }
    }
  }

  static ui void drawText(string text, int xCenter, int y)
  {
    Font aFont = Font.findFont("NewSmallFont");
    int x = xCenter - aFont.stringWidth(text) / 2;
    Screen.drawText(aFont,
                    Font.CR_Gray,
                    x,
                    y,
                    text,
                    DTA_ScaleX,
                    1.0,
                    DTA_ScaleY,
                    1.0,
                    DTA_Alpha,
                    1.0);
  }
}

class ts_ClassToggles : OptionMenu
{
  void fill(ts_Classes classes)
  {
    mDesc.mItems.clear();
    int classesCount = classes.classes.size();
    for (int i = 0; i < classesCount; ++i)
    {
      let toggle = new("ts_ClassToggle");
      toggle.initialize(classes.classes[i]);
      mDesc.mItems.push(toggle);
    }
    mDesc.mSelectedItem = 0;
  }
}

class ts_ClassToggle : OptionMenuItemOptionBase
{
  void initialize(string aClass)
  {
    mClass = aClass;
    Super.init("$" .. aClass, '', "OnOff", NULL, 0);
  }

  override int getSelection()
  {
    return isEnabled(mClass);
  }

  static clearscope bool isEnabled(string className)
  {
    let toggles = Dictionary.fromString(Cvar.findCvar("ts_classes").getString());
    string value = toggles.at(className);
    bool isEnabled = value.length() == 0 ? true : value.toInt();
    return isEnabled;
  }

  override void setSelection(int selection)
  {
    let cvar = Cvar.findCvar("ts_classes");
    let toggles = Dictionary.fromString(cvar.getString());
    toggles.insert(mClass, string.format("%d", selection));
    cvar.setString(toggles.toString());
  }

  private string mClass;
}
